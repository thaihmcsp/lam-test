  ....
}, {collection:'name'})                       // đặt tên bảng

const UserModel = mongoose.model('user', UserSchema)  // tạo công cụ để làm việc với bảng data

model.create({                                // thêm mới data
  data                                        // tạo dataBase, bảng nếu chưa có
})                                            // trả về promise

model.find({                                  // trả về array kết quả
  đk tìm kiếm                                 // trả về promise
})

model.findOne({                               // trả về kết quả đầu tiên thỏa mãn
    đk tìm kiếm                               // trả về promise
})

model.updateOne({                             // sửa data đầu tiên thỏa mãn  
    đk tìm kiếm                               // trả về promise
},{                                           // trả về báo cáo 
    data mới
})

model.deleteOne({                             // xóa data đầu tiên thỏa mãn  
    đk tìm kiếm                               // trả về promise
})                                            // trả về báo cáo

$lt         <
$lte        <=
$gt         >
$gte        >=

$ne         !==

$in         tìm kiếm những data thỏa mãn bất kỳ phần tử nào

$nin        not in - ngược lại với in
            tìm kiếm những data khác tất cả phần tử bên trong


.limit(x)   giới hạn trả về tối đa x phần tử

.skip(n)    bỏ qua n phần tử

.sort()     xắp xếp data
            1: chiều thuận
            -1 chiều ngược
            có thể sort theo nhiều đk

  ref: 'tên bảng data'
}

B2: dùng populate để kích hoạt tìm kiếm tham chiếu

model.find(đk).populate( field ) 

tìm kiếm tham chiếu theo field đã setup trong model

populate(field)           populate array, string

populate(field.key)       populate array chứa object, object
